var pin;
var crcl;
var latlng;
function initMap(elem_id, latlng_init) {
    if (!latlng_init) latlng_init = {lat: 51.109373, lng: 17.085701};
    var map = new google.maps.Map(document.getElementById(elem_id), {
       zoom: 9,
       center: latlng_init,
       mapTypeId: google.maps.MapTypeId.TERRAIN,
       streetViewControl: false,
       mapTypeControl: true,
        mapTypeControlOptions: {
          style: google.maps.MapTypeControlStyle.HORIZONTAL_BAR ,
          mapTypeIds: [
            google.maps.MapTypeId.TERRAIN,
            google.maps.MapTypeId.HYBRID
          ]
        }
    });

    map.addListener('click', function(e){
        latlng = e.latLng;
        if (!pin) {

            pin = new google.maps.Marker({
                position: e.latLng,
                draggable:true,
                map: map,
            });

            crcl = new google.maps.Circle({
                strokeColor: '#8833ff',
                strokeOpacity: 0.3,
                strokeWeight: 1.5,
                fillColor: '#8833ff',
                fillOpacity: 0.15,
                map: map,
                center:  latlng,
                radius: 60*1000,
                // draggable: true,
                clickable: false
            });


            pin.addListener('dragend', function(e){
                latlng = e.latLng;
                crcl.setCenter(latlng);
                map.panTo(latlng);
            });
        }
        crcl.setCenter(latlng);
        pin.setPosition(latlng);
        map.panTo(latlng);
    });

    return map;

}
